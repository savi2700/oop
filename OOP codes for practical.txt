#include<iostream>
#include<conio.h>
using namespace std;
class integer
{int m,n;
public:
	integer()
	{
		m=0;
		n=0;
	}
	integer(int a,int b)
	{
		m=a;
		n=b;
	}
	integer(integer &i)
	{
		m=i.m;
		n=i.n;
	}
	void display()
	{cout<<"\nm="<<m<<endl;
	 cout<<"n="<<n<<endl;
	}
};
int main()
{integer i1;
integer i2(10,20);
integer i3(i2);
cout<<"\nobject i1";
i1.display();
cout<<"\nobject i2";
i2.display();
cout<<"\nobject i3";
i3.display();
getch();
}
